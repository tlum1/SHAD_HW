def bins(p, q, n_bins=10):
    breakpoints = np.arange(0, n_bins + 1) / (n_bins) * 100
    breakpoints = np.histogram(p, n_bins)[1]

    p_percents = np.histogram(p, breakpoints)[0] / len(p)
    q_percents = np.histogram(q, breakpoints)[0] / len(q)

    p_percents = np.clip(p_percents, a_min=0.0001, a_max=None)
    q_percents = np.clip(q_percents, a_min=0.0001, a_max=None)

    return p_percents, q_percents

def kl(a, b):
    a, b = bins(a, b)

    return np.sum(np.where(a != 0, a * np.log(a / b), 0))

def psi(p: np.ndarray, q: np.ndarray):
    return kl(p, q) + kl(q, p)
