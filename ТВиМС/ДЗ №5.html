Асимпотитечкий доверительный интервал¶

$$\bar{x} \pm z_{1 - \frac{\alpha}{2}} . \sqrt{\frac{\bar{s}^2}{n}}$$

def asymptotic_confidence_interval(sample, n_obs, alpha=0.05) -> list[float]:
    """
    Строит 1 - alpha асимптотический доверительный интервал
    """
    mean = sample.mean()
    std = sample.std()
    z_crit = stats.norm(0, 1).ppf(1 - alpha / 2)
    left = mean - z_crit * (sample.std() / (n_obs**0.5))
    right = mean + z_crit * (sample.std() / (n_obs**0.5))

    return [left, right]

def accurate_confidence_interval(sample, n_obs, alpha=0.05) -> list[float]:
    """
    Строит 1 - alpha точный доверительный интервал
    """
    mean = sample.mean()
    std = sample.std()
    t_rv =stats.t(n_obs - 1)
    t_crit = t_rv.ppf(1 - alpha / 2)

    left = mean - t_crit*(std / (n_obs**0.5))
    right = mean + t_crit*(std / (n_obs**0.5))

    return [left, right]

def bootstrap_confidence_interval(sample, stat, alpha=0.05):
    """
    Стоит эфронов 1 - alpha доверительный интервал для выбранной статистики stat
    """
    n = sample.shape[0]
    distribution_lst = []
    for _ in range(100):
        a_ = np.random.choice(sample, size=n, replace=True)
        distribution_lst.append(stat(a_))

    return np.percentile(distribution_lst, q=[alpha / 2 * 100, (1 - alpha / 2) * 100])
